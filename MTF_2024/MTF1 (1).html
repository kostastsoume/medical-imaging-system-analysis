import numpy as np
import matplotlib.pyplot as plt
from PIL import Image
from scipy.ndimage import uniform_filter

# Φόρτωση της εικόνας quadrant phantom
phantom_path = r"C:\Users\kostas\Downloads\phantom_C9FF1H.gif"
phantom_image = Image.open(phantom_path).convert('L')  # Μετατροπή σε grayscale
phantom_array = np.array(phantom_image)

# Επιλογή μιας περιοχής ενδιαφέροντος για το προφίλ έντασης (π.χ. μια στήλη στο κέντρο)
x_center = phantom_array.shape[1] // 2
profile_intensity = phantom_array[:, x_center]

# Υπολογισμός των Pmax και Pmin
Pmax = np.max(profile_intensity)
Pmin = np.min(profile_intensity)
MTF_value = (Pmax - Pmin) / (Pmax + Pmin)

# Σχεδίαση του προφίλ έντασης
plt.figure(figsize=(10, 5))
plt.plot(profile_intensity, label='Προφίλ έντασης')
plt.axhline(y=Pmax, color='r', linestyle='--', label=f'Pmax: {Pmax}')
plt.axhline(y=Pmin, color='b', linestyle='--', label=f'Pmin: {Pmin}')
plt.title('Εξαγωγή προφίλ έντασης MTF')
plt.xlabel('Θέση στην εικόνα')
plt.ylabel('Ένταση')
plt.legend()
plt.show()

# --- Ανάλυση Ανομοιογένειας Πεδίου ---

# Φόρτωση της εικόνας ανομοιογένειας
homogeneity_path = r"C:\Users\kostas\Downloads\homogeneity_image_C9FF1A.gif"
homogeneity_image = Image.open(homogeneity_path).convert('L')
homogeneity_array = np.array(homogeneity_image)

# Υπολογισμός της μέσης τιμής σε παράθυρο 13x13
filtered_image = uniform_filter(homogeneity_array, size=13)

# Εύρεση Amax και Amin
Amax = np.max(filtered_image)
Amin = np.min(filtered_image)
homogeneity_ratio = (Amax - Amin) / Amax

# Εντοπισμός των σημείων Amax και Amin
amin_coords = np.where(filtered_image == Amin)
amax_coords = np.where(filtered_image == Amax)

# Σχεδίαση της εικόνας με τις ενδείξεις
plt.figure(figsize=(10, 5))
plt.imshow(homogeneity_array, cmap='gray')
plt.scatter(amin_coords[1], amin_coords[0], color='blue', label='Amin', marker='o')
plt.scatter(amax_coords[1], amax_coords[0], color='red', label='Amax', marker='x')
plt.title('Ανάλυση Ανομοιογένειας Πεδίου')
plt.legend()
plt.show()

print(f"Amax: {Amax}, Amin: {Amin}, Homogeneity Ratio: {homogeneity_ratio:.4f}")
